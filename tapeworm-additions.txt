// Returns the chapter object for a given wave number
var getChapterForWave = function(waveNum) {
  for (var i = 0; i < CHAPTERS.length; i++) {
    if (waveNum >= CHAPTERS[i].waves[0] && waveNum <= CHAPTERS[i].waves[1]) {
      return CHAPTERS[i];
    }
  }
  return CHAPTERS[CHAPTERS.length - 1];
};

// Returns a lore text to show between waves (or null if none)
var getLoreText = function(waveNum) {
  var chapter = getChapterForWave(waveNum);
  var waveInChapter = waveNum - chapter.waves[0];
  // Show lore at specific intervals within each chapter
  if (waveInChapter < chapter.loreTexts.length) {
    return chapter.loreTexts[waveInChapter];
  }
  return null;
};

// Returns chapter intro text when entering a new chapter
var getChapterIntro = function(waveNum) {
  for (var i = 0; i < CHAPTERS.length; i++) {
    if (waveNum === CHAPTERS[i].waves[0]) {
      return "CHAPTER " + CHAPTERS[i].id + ": " + CHAPTERS[i].name.toUpperCase() + "\n" + CHAPTERS[i].description;
    }
  }
  return null;
};

// Update function for the Kiwi boss — call this inside the enemy update loop
// when e.behavior === 'kiwi_boss'
var updateKiwiBoss = function(e, dt) {
  var hpPercent = e.hp / e.maxHp;

  // Phase check
  if (hpPercent <= 0.5 && e.phase === 1) {
    e.phase = 2;
    e.furNeedleCooldown = 0.4; // Faster shooting in phase 2
    e.speed = 60; // Faster movement
    // Flash effect to signal phase change
    spawnParticles(e.x, e.y, "#ff4444", 30, 200, 0.8);
    spawnParticles(e.x, e.y, "#6b8e23", 20, 150, 0.6);
  }

  // --- ATTACK 1: Fur needle spiral ---
  e.furNeedleTimer -= dt;
  e.spiralAngle += dt * (e.phase === 2 ? 4 : 2);
  if (e.furNeedleTimer <= 0) {
    var needleCount = e.phase === 2 ? 8 : 5;
    for (var i = 0; i < needleCount; i++) {
      var a = e.spiralAngle + (Math.PI * 2 / needleCount) * i;
      var speed = 200 + (e.phase === 2 ? 80 : 0);
      enemyProjectiles.push({
        x: e.x,
        y: e.y,
        vx: Math.cos(a) * speed,
        vy: Math.sin(a) * speed,
        damage: 8 + wave,
        radius: 4,
        life: 2.5,
        color: "#8B4513"
      });
    }
    e.furNeedleTimer = e.furNeedleCooldown;
  }

  // --- ATTACK 2: Acid seed bombs ---
  e.seedBombTimer -= dt;
  if (e.seedBombTimer <= 0) {
    var bombCount = e.phase === 2 ? 5 : 3;
    for (var j = 0; j < bombCount; j++) {
      var bx = player.x + rng(-80, 80);
      var by = player.y + rng(-80, 80);
      // Create acid zone after a delay (using acidTrails)
      for (var k = 0; k < 6; k++) {
        acidTrails.push({
          x: bx + rng(-20, 20),
          y: by + rng(-20, 20),
          life: 4
        });
      }
      // Visual: seed projectile toward the zone
      spawnParticles(bx, by, "#556b2f", 8, 60, 0.3);
    }
    e.seedBombTimer = e.phase === 2 ? 2 : 3;
  }

  // --- ATTACK 3: Charge at player ---
  if (!e.isCharging) {
    e.chargeTimer -= dt;
    if (e.chargeTimer <= 0) {
      e.isCharging = true;
      e.chargeTargetX = player.x;
      e.chargeTargetY = player.y;
      // Brief telegraph: flash white
      e.flashTimer = 0.5;
    }
  } else {
    // Charge toward saved target position
    var ca = Math.atan2(e.chargeTargetY - e.y, e.chargeTargetX - e.x);
    var chargeSpeed = 350;
    e.vx = Math.cos(ca) * chargeSpeed;
    e.vy = Math.sin(ca) * chargeSpeed;
    e.x += e.vx * dt;
    e.y += e.vy * dt;
    // Check if reached target or close enough
    var distToTarget = Math.sqrt(
      (e.x - e.chargeTargetX) * (e.x - e.chargeTargetX) +
      (e.y - e.chargeTargetY) * (e.y - e.chargeTargetY)
    );
    if (distToTarget < 20) {
      e.isCharging = false;
      e.chargeTimer = e.phase === 2 ? 3 : 5;
      // Impact shockwave
      spawnParticles(e.x, e.y, "#6b8e23", 15, 180, 0.5);
    }
  }

  // --- PHASE 2: Spawn mini kiwi seeds ---
  if (e.phase === 2) {
    e.miniKiwiTimer -= dt;
    if (e.miniKiwiTimer <= 0) {
      for (var m = 0; m < 3; m++) {
        var miniKiwi = {
          type: "mini_kiwi",
          x: e.x + rng(-30, 30),
          y: e.y + rng(-30, 30),
          vx: 0,
          vy: 0,
          hp: 20,
          maxHp: 20,
          radius: 10,
          speed: 90,
          xp: 3,
          coins: 1,
          behavior: "chase",
          shootTimer: 0,
          trailTimer: 0,
          bounceAngle: rng(0, Math.PI * 2),
          swarmOffset: rng(0, Math.PI * 2),
          flashTimer: 0,
          spawnTimer: 0
        };
        enemies.push(miniKiwi);
      }
      e.miniKiwiTimer = 4;
    }
  }

  // Normal movement (when not charging)
  if (!e.isCharging) {
    var pa = angle(e, player);
    e.vx = Math.cos(pa) * e.speed;
    e.vy = Math.sin(pa) * e.speed;
    e.x += e.vx * dt;
    e.y += e.vy * dt;
  }

  e.x = clamp(e.x, ARENA.x, ARENA.x + ARENA.w);
  e.y = clamp(e.y, ARENA.y, ARENA.y + ARENA.h);
};

// Draw function for the Kiwi boss — call this in the enemy draw section
// when e.type === 'kiwi'
var drawKiwiBoss = function(e) {
  var hpPercent = e.hp / e.maxHp;
  var ctx = document.getElementById("c").getContext("2d");

  // Outer fuzzy border (brown fur)
  var furRadius = e.radius + 8;
  ctx.save();
  ctx.beginPath();
  // Draw fuzzy edge with jagged circle
  var segments = 24;
  for (var i = 0; i <= segments; i++) {
    var a = (Math.PI * 2 / segments) * i;
    var jitter = Math.sin(a * 8 + Date.now() * 0.005) * 4 + 4;
    var r = furRadius + jitter;
    var fx = e.x + Math.cos(a) * r;
    var fy = e.y + Math.sin(a) * r;
    if (i === 0) {
      ctx.moveTo(fx, fy);
    } else {
      ctx.lineTo(fx, fy);
    }
  }
  ctx.closePath();
  ctx.fillStyle = "#8B4513";
  ctx.fill();

  // Inner body — green, gets redder as HP drops
  var redShift = Math.floor((1 - hpPercent) * 180);
  var greenBase = Math.floor(142 * hpPercent);
  var bodyColor = "rgb(" + (107 + redShift) + ", " + (greenBase + 35) + ", 35)";
  ctx.beginPath();
  ctx.arc(e.x, e.y, e.radius, 0, Math.PI * 2);
  ctx.fillStyle = bodyColor;
  ctx.fill();

  // Seed dots on body
  ctx.fillStyle = "#1a1a0a";
  var seedCount = 8;
  for (var s = 0; s < seedCount; s++) {
    var sa = (Math.PI * 2 / seedCount) * s + 0.3;
    var sr = e.radius * 0.5;
    ctx.beginPath();
    ctx.ellipse(
      e.x + Math.cos(sa) * sr,
      e.y + Math.sin(sa) * sr,
      3, 2, sa, 0, Math.PI * 2
    );
    ctx.fill();
  }

  // Angry face
  var eyeOffY = -8;
  var eyeSpread = 12;

  // Eyes (white with red pupils when in phase 2)
  ctx.fillStyle = "#fff";
  ctx.beginPath();
  ctx.arc(e.x - eyeSpread, e.y + eyeOffY, 7, 0, Math.PI * 2);
  ctx.fill();
  ctx.beginPath();
  ctx.arc(e.x + eyeSpread, e.y + eyeOffY, 7, 0, Math.PI * 2);
  ctx.fill();

  // Pupils
  ctx.fillStyle = e.phase === 2 ? "#ff0000" : "#111";
  ctx.beginPath();
  ctx.arc(e.x - eyeSpread + 1, e.y + eyeOffY + 1, 4, 0, Math.PI * 2);
  ctx.fill();
  ctx.beginPath();
  ctx.arc(e.x + eyeSpread + 1, e.y + eyeOffY + 1, 4, 0, Math.PI * 2);
  ctx.fill();

  // Angry eyebrows
  ctx.strokeStyle = e.phase === 2 ? "#ff0000" : "#333";
  ctx.lineWidth = 3;
  ctx.beginPath();
  ctx.moveTo(e.x - eyeSpread - 8, e.y + eyeOffY - 10);
  ctx.lineTo(e.x - eyeSpread + 4, e.y + eyeOffY - 6);
  ctx.stroke();
  ctx.beginPath();
  ctx.moveTo(e.x + eyeSpread + 8, e.y + eyeOffY - 10);
  ctx.lineTo(e.x + eyeSpread - 4, e.y + eyeOffY - 6);
  ctx.stroke();

  // Angry mouth
  ctx.strokeStyle = "#333";
  ctx.lineWidth = 2;
  ctx.beginPath();
  ctx.arc(e.x, e.y + 12, 10, 0.2, Math.PI - 0.2);
  ctx.stroke();

  // Phase 2: red aura
  if (e.phase === 2) {
    ctx.beginPath();
    ctx.arc(e.x, e.y, e.radius + 20, 0, Math.PI * 2);
    ctx.strokeStyle = "rgba(255, 0, 0, " + (0.3 + Math.sin(Date.now() * 0.01) * 0.2) + ")";
    ctx.lineWidth = 3;
    ctx.stroke();
  }

  // HP bar above boss
  var barW = 80;
  var barH = 6;
  var barX = e.x - barW / 2;
  var barY = e.y - e.radius - 25;
  ctx.fillStyle = "rgba(0,0,0,0.6)";
  ctx.fillRect(barX - 1, barY - 1, barW + 2, barH + 2);
  ctx.fillStyle = hpPercent > 0.5 ? "#6b8e23" : (hpPercent > 0.25 ? "#e67e22" : "#e74c3c");
  ctx.fillRect(barX, barY, barW * hpPercent, barH);

  // Boss name
  ctx.fillStyle = "#fff";
  ctx.font = "bold 11px sans-serif";
  ctx.textAlign = "center";
  ctx.fillText("🥝 KIWI THE DESTROYER", e.x, barY - 6);
  ctx.textAlign = "start";

  ctx.restore();
};

var MusicSystem = {
  ctx: null,
  currentLoop: null,
  masterGain: null,
  oscillators: [],
  intervalId: null,

  init: function() {
    if (!this.ctx) {
      this.ctx = new (window.AudioContext || window.webkitAudioContext)();
      this.masterGain = this.ctx.createGain();
      this.masterGain.gain.value = 0.12;
      this.masterGain.connect(this.ctx.destination);
    }
  },

  stopAll: function() {
    for (var i = 0; i < this.oscillators.length; i++) {
      try { this.oscillators[i].stop(); } catch(ex) {}
    }
    this.oscillators = [];
    if (this.intervalId) {
      clearInterval(this.intervalId);
      this.intervalId = null;
    }
    this.currentLoop = null;
  },

  // Create a persistent oscillator that plays until stopped
  _makeOsc: function(freq, type, gainVal) {
    var osc = this.ctx.createOscillator();
    var gain = this.ctx.createGain();
    osc.type = type || "sine";
    osc.frequency.value = freq;
    gain.gain.value = gainVal || 0.1;
    osc.connect(gain);
    gain.connect(this.masterGain);
    osc.start();
    this.oscillators.push(osc);
    return { osc: osc, gain: gain };
  },

  // Create a note that plays for a duration then fades
  _playNote: function(freq, dur, type, vol) {
    if (!this.ctx) return;
    var osc = this.ctx.createOscillator();
    var gain = this.ctx.createGain();
    osc.type = type || "sine";
    osc.frequency.value = freq;
    gain.gain.value = vol || 0.08;
    gain.gain.exponentialRampToValueAtTime(0.001, this.ctx.currentTime + dur);
    osc.connect(gain);
    gain.connect(this.masterGain);
    osc.start();
    osc.stop(this.ctx.currentTime + dur);
  },

  // --- GAMEPLAY MUSIC: Bass-heavy, pulsing, ominous body interior ---
  startGameplayMusic: function() {
    this.init();
    if (this.currentLoop === "gameplay") return;
    this.stopAll();
    this.currentLoop = "gameplay";

    var self = this;
    // Persistent low drone
    var drone = this._makeOsc(55, "sine", 0.07);  // Low A drone
    var drone2 = this._makeOsc(82.5, "sine", 0.04); // Low E fifth

    // Subtle LFO on the drone for "breathing" effect
    var lfo = this.ctx.createOscillator();
    var lfoGain = this.ctx.createGain();
    lfo.type = "sine";
    lfo.frequency.value = 0.5; // Slow pulse — like a heartbeat
    lfoGain.gain.value = 15;
    lfo.connect(lfoGain);
    lfoGain.connect(drone.osc.frequency);
    lfo.start();
    this.oscillators.push(lfo);

    // Rhythmic bass pulse
    var bassNotes = [55, 55, 65.4, 55, 55, 49, 55, 55];
    var beatIndex = 0;
    var bpm = 80;
    var beatInterval = 60000 / bpm;

    this.intervalId = setInterval(function() {
      if (self.currentLoop !== "gameplay") return;
      var note = bassNotes[beatIndex % bassNotes.length];
      self._playNote(note, 0.3, "triangle", 0.06);
      // Heartbeat thump on beats 0 and 2
      if (beatIndex % 4 === 0 || beatIndex % 4 === 2) {
        self._playNote(40, 0.15, "sine", 0.1);
        // Double thump (heartbeat)
        setTimeout(function() {
          self._playNote(35, 0.1, "sine", 0.07);
        }, 120);
      }
      // Eerie high note every 8 beats
      if (beatIndex % 8 === 0) {
        self._playNote(440 + Math.random() * 60, 0.8, "sine", 0.02);
      }
      beatIndex++;
    }, beatInterval);
  },

  // --- BOSS MUSIC: Faster, intense, dissonant ---
  startBossMusic: function() {
    this.init();
    if (this.currentLoop === "boss") return;
    this.stopAll();
    this.currentLoop = "boss";

    var self = this;
    // Dissonant drone
    var drone1 = this._makeOsc(55, "sawtooth", 0.04);
    var drone2 = this._makeOsc(58.27, "sawtooth", 0.03); // Bb - dissonant against A

    // Aggressive LFO
    var lfo = this.ctx.createOscillator();
    var lfoGain = this.ctx.createGain();
    lfo.type = "square";
    lfo.frequency.value = 3;
    lfoGain.gain.value = 20;
    lfo.connect(lfoGain);
    lfoGain.connect(drone1.osc.frequency);
    lfo.start();
    this.oscillators.push(lfo);

    var bassNotes = [55, 58.27, 55, 51.9, 55, 58.27, 61.7, 55];
    var beatIndex = 0;
    var bpm = 140;
    var beatInterval = 60000 / bpm;

    this.intervalId = setInterval(function() {
      if (self.currentLoop !== "boss") return;
      var note = bassNotes[beatIndex % bassNotes.length];
      self._playNote(note, 0.2, "sawtooth", 0.05);
      // Aggressive percussion hits
      if (beatIndex % 2 === 0) {
        self._playNote(80, 0.08, "square", 0.08);
      }
      // Dissonant stabs every 4 beats
      if (beatIndex % 4 === 0) {
        self._playNote(note * 4 + 5, 0.15, "square", 0.03);
        self._playNote(note * 5 + 7, 0.12, "sawtooth", 0.02);
      }
      // Rapid hi notes in phase transitions
      if (beatIndex % 8 >= 6) {
        self._playNote(880 + Math.random() * 200, 0.05, "square", 0.02);
      }
      beatIndex++;
    }, beatInterval);
  },

  // --- SHOP MUSIC: Chill, elevator-music style ---
  startShopMusic: function() {
    this.init();
    if (this.currentLoop === "shop") return;
    this.stopAll();
    this.currentLoop = "shop";

    var self = this;
    // Smooth pad
    var pad1 = this._makeOsc(220, "sine", 0.03);
    var pad2 = this._makeOsc(277.18, "sine", 0.02); // C#4 — major third
    var pad3 = this._makeOsc(329.63, "sine", 0.02); // E4 — fifth

    // Gentle LFO vibrato
    var lfo = this.ctx.createOscillator();
    var lfoGain = this.ctx.createGain();
    lfo.type = "sine";
    lfo.frequency.value = 0.3;
    lfoGain.gain.value = 3;
    lfo.connect(lfoGain);
    lfoGain.connect(pad1.osc.frequency);
    lfo.start();
    this.oscillators.push(lfo);

    // Jazz-ish melody notes
    var melodyNotes = [
      440, 494, 523, 587, 523, 494, 440, 392,
      440, 523, 587, 659, 587, 523, 494, 440,
      392, 440, 494, 523, 440, 392, 349, 392
    ];
    var noteIndex = 0;
    var bpm = 100;
    var beatInterval = 60000 / bpm;

    this.intervalId = setInterval(function() {
      if (self.currentLoop !== "shop") return;
      var note = melodyNotes[noteIndex % melodyNotes.length];
      // Smooth sine melody
      self._playNote(note, 0.5, "sine", 0.04);
      // Gentle bass
      if (noteIndex % 4 === 0) {
        self._playNote(note / 4, 0.8, "sine", 0.05);
      }
      // Soft chime every 8 notes
      if (noteIndex % 8 === 0) {
        self._playNote(note * 2, 0.6, "sine", 0.015);
        self._playNote(note * 2.5, 0.6, "sine", 0.01);
      }
      noteIndex++;
    }, beatInterval);
  },

  // --- VICTORY JINGLE ---
  playVictoryJingle: function() {
    this.init();
    this.stopAll();
    this.currentLoop = "victory";

    var self = this;
    var notes = [
      { f: 523.25, t: 0,    d: 0.2 }, // C5
      { f: 587.33, t: 0.15, d: 0.2 }, // D5
      { f: 659.25, t: 0.3,  d: 0.2 }, // E5
      { f: 783.99, t: 0.5,  d: 0.3 }, // G5
      { f: 659.25, t: 0.7,  d: 0.15 }, // E5
      { f: 783.99, t: 0.85, d: 0.5 }, // G5
      { f: 1046.5, t: 1.1,  d: 0.7 }, // C6 (big finish)
      // Harmony
      { f: 523.25, t: 1.1,  d: 0.7 }, // C5
      { f: 659.25, t: 1.1,  d: 0.7 }, // E5
    ];

    for (var i = 0; i < notes.length; i++) {
      (function(note) {
        setTimeout(function() {
          self._playNote(note.f, note.d, "sine", 0.06);
          // Add shimmer
          self._playNote(note.f * 2, note.d * 0.7, "sine", 0.02);
        }, note.t * 1000);
      })(notes[i]);
    }

    // Sparkle trail
    for (var j = 0; j < 6; j++) {
      (function(idx) {
        setTimeout(function() {
          self._playNote(1200 + idx * 200, 0.15, "sine", 0.015);
        }, 1400 + idx * 80);
      })(j);
    }
  },

  // --- DEATH JINGLE ---
  playDeathJingle: function() {
    this.init();
    this.stopAll();
    this.currentLoop = "death";

    var self = this;
    var notes = [
      { f: 392, t: 0,    d: 0.3 }, // G4
      { f: 349, t: 0.25, d: 0.3 }, // F4
      { f: 311, t: 0.5,  d: 0.3 }, // Eb4
      { f: 261, t: 0.8,  d: 0.8 }, // C4 (sad finish)
      // Dissonant harmony
      { f: 277, t: 0.8,  d: 0.8 }, // C#4
    ];

    for (var i = 0; i < notes.length; i++) {
      (function(note) {
        setTimeout(function() {
          self._playNote(note.f, note.d, "sawtooth", 0.04);
          self._playNote(note.f * 0.5, note.d, "sine", 0.03);
        }, note.t * 1000);
      })(notes[i]);
    }
  }
